// FUNGSI KOMPRESI YANG LEBIH BANDEL & ANTI-MACET async function compressImg(file) { return new Promise((resolve) => { const img = new Image(); img.src = URL.createObjectURL(file); // Safety Timeout: Jika 5 detik tidak respon, kirim file asli const timeout = setTimeout(() => { console.log("Compression timeout, using original file"); resolve(file); }, 5000); img.onload = () => { clearTimeout(timeout); const canvas = document.createElement('canvas'); const ctx = canvas.getContext('2d'); let w = img.width, h = img.height, max = 800; if (w > h) { if (w > max) { h *= max / w; w = max; } } else { if (h > max) { w *= max / h; h = max; } } canvas.width = w; canvas.height = h; ctx.drawImage(img, 0, 0, w, h); canvas.toBlob((blob) => resolve(blob || file), 'image/jpeg', 0.7); }; img.onerror = () => { clearTimeout(timeout); resolve(file); // Jika error, balikkan file asli }; }); } window.saveToFirebase = async () => { const input = (id) => document.getElementById(id).value; const fileInput = document.getElementById('fileGambar'); const file = fileInput.files[0]; const load = document.getElementById('loadingOverlay'); if(!input('nama') || !file) return alert("Nama & Foto harus ada, Boss!"); load.classList.remove('hidden'); // Munculkan loading try { // Jalankan Kompresi const blob = await compressImg(file); // Upload ke Firebase Storage const sRefImg = sRef(storage, `produk/img_${Date.now()}.jpg`); const uploadTask = await uploadBytes(sRefImg, blob); const url = await getDownloadURL(uploadTask.ref); // Simpan ke Database await push(ref(db, 'produk'), { nama: input('nama'), barcode: input('barcode') || "NOCODE", harga: parseInt(input('harga')) || 0, stokDasar: parseInt(input('stokDasar')) || 0, stokDetail: input('stokDetail') || "", kategori: document.getElementById('kategori').value, gambar: url, timestamp: Date.now() }); alert("Berhasil Disimpan! ⚡"); location.reload(); // Refresh halaman } catch (e) { console.error(e); alert("Gagal upload: " + e.message); } finally { load.classList.add('hidden'); // Pastikan loading HILANG apapun yang terjadi } };
